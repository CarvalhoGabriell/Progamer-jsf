pipeline {
    agent any
    triggers {
        pollSCM 'H/2 * * * *'
    }
    stages {
        stage('Código git') {
            steps {
                echo 'Clonando...'
                git branch: 'main', url: 'https://github.com/CarvalhoGabriell/progamer-jsf.git'
            }
        }
        stage('Build') {
            steps {
                withSonarQubeEnv('SonarQube') {
                    echo 'Build...'
                    sh '/opt/maven/bin/mvn clean package sonar:sonar'
                }
            }
        }
        stage('Test') {
            steps {
                echo 'Teste...'
                sh '/opt/maven/bin/mvn test'
            }
        }
        stage('Install') {
            steps {
                echo 'Instalando...'
                sh '/opt/maven/bin/mvn install'
            }
        }
        stage('Compilação') {
            steps{
                echo 'Compilando...'
                sh "/opt/maven/bin/mvn compiler:compile"
            }
        }
        stage('Aceite') {
            steps {
                echo 'Espere o aceite de usuário'
                input(message: 'Mandar para produção?', ok: 'Sim')
            }
        }
        stage('Produção') {
            steps {
                echo 'Rodando'
            }
        }
        stage('Jboss') {
            steps{
                //echo 'Deploy...'
                sh "/opt/maven/bin/mvn dependency:tree"
        }
    }   
        stage('Deploy') {
            steps{
                echo 'Deploy...'
                sh "/opt/maven/bin/mvn deploy"
        }
    }
    }
}
    stage("Quality Gate") { 
    def qualityGate = waitForQualityGate() 
      if (qualityGate.status != 'OK') {
        error "O código não está de acordo com as regras do Sonar: ${qualityGate.status}"
      }
      else {
          echo "Código está de acordo com as regras do Sonar."
      }
}
